Arduino Code For Instrument

This is the code Jaeho and I have made to create a very simple arduino version of traffic lights. The only change we have made is to have green lights on for 10 seconds instead of 35 seconds to speed up the process. 

  int buttonA = 2;
  int buttonB = 3;
  int buttonC = 4;
  int buttonD = 5;
  int buttonE = 6;
  int buttonF = 7;
  int buttonG = 8;

void setup() {
pinMode (buttonA, OUTPUT);
pinMode (buttonB, OUTPUT);
pinMode (buttonC, OUTPUT);
pinMode (buttonD, OUTPUT);
pinMode (buttonE, OUTPUT);
pinMode (buttonF, OUTPUT);
pinMode (buttonG, OUTPUT);
}

void loop() {
  if (digitalRead(buttonA)==HIGH) {
    c ();
  }
  if (digitalRead(buttonB)==HIGH) {
    d ();
  }
  if (digitalRead(buttonC)==HIGH) {
    e ();
  }
  if (digitalRead(buttonD)==HIGH) {
    f ();
  }
  if (digitalRead(buttonE)==HIGH) {
    g ();
  }
  if (digitalRead(buttonF)==HIGH) {
    a ();
  }
  if (digitalRead(buttonG)==HIGH) {
    b ();
  }
}
void a () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 440, 30);
}
void b () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 494, 30);
}
void c () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 523, 30);
}
void d () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 587, 30);
}
void e () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 659, 30);
}
void f () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 698, 30);
}
void g () {
  // play the tone using pin 13 with frequency 415 Hz for 500 ms = 0.5 seconds
  tone(13, 784, 30);
}
